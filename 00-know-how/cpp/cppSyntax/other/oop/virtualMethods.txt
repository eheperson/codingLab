#include <iostream>

class Vehicle
{
public:
    virtual void PrintFunction();
    
};

void Vehicle::PrintFunction()
{
    std::cout << "PRINT BASe" << std::endl;
}

class BMW : public Vehicle
{
public:
    void PrintFunction();
    
};

void BMW::PrintFunction()
{
    std::cout << "PRINT BMW" << std::endl;
}

int main(int argc, const char * argv[])
{
    Vehicle v1;
    
    v1.PrintFunction();
    
    BMW b1;
    b1.PrintFunction();
    
    return 0;
}
