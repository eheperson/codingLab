#include <iostream>
#include <fstream>

int main(int argc, const char * argv[])
{
    std::ofstream file ("textfile.txt");
    
    if ( file.is_open() )
    {
        file << "We have added a file\n";
        file.close();
    }
    else
    {
        std::cout << "Unable to open file" << std::endl;
    }
    
    return 0;
}
